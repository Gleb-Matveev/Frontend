void QuickSort(int *a, int n)
{
  static int x;
  int b = 0, e = n - 1;

  if (n < 2)
    return;

  x = a[n / 2];
  while(b <= e)
  {  
    while (a[b] < x)
      b++;
    while (a[e] > x)
      e--;
    
    if (b <= e)
    {
      if (b != e)
        Swap(&a[b], &a[e]);
      b++;
      e--;
    }
  }
  QuickSort(a, e + 1);
  QuickSort(a + b, n - b);
}
